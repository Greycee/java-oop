package bookProject;
public class Book implements Publication  {
  private String title, author
  private int totPages
  private int currentPage
  private boolean open
  private Person reader

  public void details() {
    // we can erase everything and generate a new method (right click) called toString. It will generate:
    @Override
    public String toString() {
      return "Book{"title=" + title 
      + "\n author=" + author
      + "\n totPages=" + totPages
      + "\n currentPage=" + currentPage
      + "\n open=" + open
      + "\n reader=" + reader.getName() + '}'
      }
    // we can remove the override and rename the toString method to the name we want (details)
    }
  }

  public Book(String title, String author, int totPages, Person reader) {
    this.title = title
    this.author =  author
    this.totPages = totPages
    this.open = false
    this.currentPage = 0
    this.reader = reader
  }

  public String getTitle() {
    return title
  }

  public void setTitle(String t){
    this.title = t
  }

  public String getAuthor() {
    return author
  }

  public void setAuthor(String a){
    this.author = a
  }

  public boolean getTotPages() {
    return totPages
  }

  public void setTotPages(int tp){
    this.totPages = tp
  }

  public boolean getCurrentPage() {
    return currentPage
  }

  public void setCurrentPage(int cp){
    this.currentPage = cp
  }

  public boolean isOpen() {
    return open
  }

  public void setOpen(boolean o){
    this.open = o
  }

  public Person getReader() {
    return reader
  }

  public Person setReader(String r){
    this.reader = r
  }

  @Override
  public void open() {
    this.open = true
  }

  @Override
  public void close() {
    this.open = false
  }

  @Override
  public void turnPage(int p) {
    if (p> this.totPages) {
      this.currentPage = 0
    } else {
      this.currentPage = p
    }
  }

  @Override
  public void forwardPage() {
    this.currentPage++
  }

  @Override
  public void previousPage() {
    this.currentPage--
  }
}
